*passenger 테이블
CREATE TABLE passenger(
no int(11) auto_increment PRIMARY KEY,
id varchar(20) UNIQUE KEY,
pwd varchar(20),
name varchar(20),
phone varchar(20)
);

*driver 테이블
CREATE TABLE driver(
no int(11) auto_increment PRIMARY KEY,
id varchar(20) UNIQUE KEY,
pwd varchar(20),
name varchar(20),
phone varchar(20),
isOn bool default false
);

*calls 테이블
CREATE TABLE calls(
num int(11) unsigned auto_increment,
calltime timestamp default NOW(),
status int(11),
sid varchar(20),
gid varchar(20),
slocLat int(11),
slocLong int(11),
sdestLat int(11),
sdestLong int(11),
smessage varchar(100),
PRIMARY KEY (num),
FOREIGN KEY (sid) REFERENCES passenger(id),
FOREIGN KEY (gid) REFERENCES driver(id)
);
